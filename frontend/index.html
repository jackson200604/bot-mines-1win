const grid = document.getElementById('grid'); let selectedTiles = []; const BACKEND_URL = "https://bot-mines-1win.onrender.com"; // Création de la grille 5x5 for (let i = 0; i < 25; i++) { const tile = document.createElement('div'); tile.className = "tile border-2 border-slate-600 bg-slate-700 rounded-md flex items-center justify-center text-xs text-slate-500"; tile.innerText = i; tile.onclick = () => { tile.classList.toggle('selected'); if (selectedTiles.includes(i)) selectedTiles = selectedTiles.filter(t => t !== i); else selectedTiles.push(i); }; grid.appendChild(tile); } async function getPrediction() { const btn = document.getElementById('btn'); btn.innerText = "Calcul en cours..."; btn.disabled = true; try { const response = await fetch(`${BACKEND_URL}/predict`, { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ history: selectedTiles, mines_count: 3 }) }); const data = await response.json(); // Reset recommandations précédentes document.querySelectorAll('.tile').forEach(t => t.classList.remove('recommended')); // Afficher les nouvelles data.recommended_tiles.forEach(index => { grid.children[index].classList.add('recommended'); }); document.getElementById('result').classList.remove('hidden'); document.getElementById('tiles-text').innerText = `Cases : ${data.recommended_tiles.join(', ')}`; document.getElementById('confidence-text').innerText = `Confiance : ${data.confidence}%`; } catch (err) { alert("Erreur de connexion au backend Render"); } finally { btn.innerText = "OBTENIR LA PRÉDICTION"; btn.disabled = false; } }
